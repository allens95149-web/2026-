      const totalSeconds = Math.floor(milliseconds / 1000);
      const minutes = Math.floor(totalSeconds / 60);
      const seconds = totalSeconds % 60;

    function initFinalsInputs(team1, team2) {
      document.getElementById('finals-team1-name').textContent = team1;
      document.getElementById('finals-team2-name').textContent = team2;
      const container1 = document.getElementById('finals-team1-inputs');
      const container2 = document.getElementById('finals-team2-inputs');
      if (!container1 || !container2) return;
      container1.innerHTML = '';
      container2.innerHTML = '';
      for (let i = 1; i <= 10; i++) {
        const createInput = (team) => {
          const input = document.createElement('input');
          input.type = 'number';
          input.className = 'threepoint-finals-input input-base text-center';
          input.placeholder = i;
          input.dataset.team = team;
          input.dataset.player = i;
          input.min = '0';
          input.disabled = false;
          input.addEventListener('input', function(e) {
            const val = parseInt(this.value) || 0;
            if (!gameData.threePoint.finals.teams[team]) {
              gameData.threePoint.finals.teams[team] = { players: Array(10).fill(0), total: 0 };
            }
            gameData.threePoint.finals.teams[team].players[i - 1] = val;
            const total = gameData.threePoint.finals.teams[team].players.reduce((sum, v) => sum + (v || 0), 0);
            gameData.threePoint.finals.teams[team].total = total;
            document.getElementById(`finals-team${team === team1 ? '1' : '2'}-total`).textContent = total;
            updatePlayerDisplay();
            saveGameData();
          });
          return input;
        };
        container1.appendChild(createInput(team1));
        container2.appendChild(createInput(team2));
      }
    }

    function finalizeFinals() {
      if (userRole !== 'staff') return;
      // ? 防止重複結算
      if (finalizationState.threePointFinalized) return;
      
      const [team1, team2] = gameData.threePoint.finals.qualified;
      const t1Score = gameData.threePoint.finals.teams[team1]?.total || 0;
      const t2Score = gameData.threePoint.finals.teams[team2]?.total || 0;

      Object.assign(gameData.teams[team1], { threePointScore: t1Score });
      Object.assign(gameData.teams[team2], { threePointScore: t2Score });
      Object.assign(gameData.teams['ABC'.replace(team1, '').replace(team2, '')], { threePointScore: 0 });

      const winner = t1Score >= t2Score ? team1 : team2;
      gameData.threePointWinner = winner;
      matchWinners.threePoint = winner;
      
      // ? 修正：驟死賽權重 - 每項僅獲得 1 局
      championWins[winner] += 1;

      winnerState.threePoint = {
        team: winner,
        score: Math.max(t1Score, t2Score)
      };

      updateTeamTotalScores();
      setActiveStep(3);
      showThreePointWinnerAnnouncement(winner, Math.max(t1Score, t2Score));
      
      // 鎖定結算狀態
      finalizationState.threePointFinalized = true;
      lockThreePointInputs();
      updateThreePointFinalizeBtnStatus();
      
      saveGameData();
    }

    // ? 哨聲特效
    function playWhistleSound() {
      const audioContext = new (window.AudioContext || window.webkitAudioContext)();
      const oscillator = audioContext.createOscillator();
      const gainNode = audioContext.createGain();
      
      oscillator.connect(gainNode);
      gainNode.connect(audioContext.destination);
      
      oscillator.frequency.value = 800;
      gainNode.gain.setValueAtTime(0.3, audioContext.currentTime);
      gainNode.gain.exponentialRampToValueAtTime(0.01, audioContext.currentTime + 0.3);
      
      oscillator.start(audioContext.currentTime);
      oscillator.stop(audioContext.currentTime + 0.3);
    }

    function finalizeMatch() {
      if (userRole !== 'staff') return;
      // ? 防止重複結算
      if (finalizationState.fiveVFiveFinalized) return;

      // ? 自動停止計時
      if (fiveVFiveState.firstHalfRunning) pauseFirstHalfTimer();
      if (fiveVFiveState.secondHalfRunning) pauseSecondHalfTimer();

      gameData.fiveVFive[fiveVFiveState.team1] = (gameData.fiveVFive[fiveVFiveState.team1] || 0) + fiveVFiveState.team1Score;
      gameData.fiveVFive[fiveVFiveState.team2] = (gameData.fiveVFive[fiveVFiveState.team2] || 0) + fiveVFiveState.team2Score;

      updateTeamTotalScores();
      updateFiveVFiveCumulativeDisplay();
      saveGameData();

      const matchOrder = ['ab', 'ac', 'bc'];
      const currentIdx = matchOrder.indexOf(fiveVFiveState.currentMatchup);
      
      if (currentIdx < matchOrder.length - 1) {
        const nextMatch = matchOrder[currentIdx + 1];
        const nextBtn = document.querySelector(`.five-v-five-matchup[data-match="${nextMatch}"]`);
        if (nextBtn) {
          fiveVFiveState.team1Score = 0;
          fiveVFiveState.team2Score = 0;
          fiveVFiveState.half = 1;
          fiveVFiveState.firstHalfTime = 600;
          fiveVFiveState.secondHalfTime = 600;
          fiveVFiveState.restingTime = false;
          nextBtn.click();
        }
      } else {
        // ? 三場五五賽結束，比對累積分數最高者給予 1 局獲勝
        const fiveVFiveScores = [
          { team: 'A', score: gameData.fiveVFive.A || 0 },
          { team: 'B', score: gameData.fiveVFive.B || 0 },
          { team: 'C', score: gameData.fiveVFive.C || 0 }
        ].sort((a, b) => b.score - a.score);

        const fiveVFiveWinner = fiveVFiveScores[0].team;
        championWins[fiveVFiveWinner] += 1;
        matchWinners.fiveVFive = fiveVFiveWinner;
        
        // 鎖定結算狀態
        finalizationState.fiveVFiveFinalized = true;
        lockFiveVFiveInputs();
        updateFiveVFiveFinalizeBtnStatus();
        
        // 顯示五五賽冠軍公告
        showFiveVFiveWinnerAnnouncement();
        setActiveStep(4);
      }
    }
